            continue

model.add(Flatten())

model.add(Dense(20,activation='relu'))
model.add(Dense(15,activation='relu'))
model.add(Dense(10,activation = 'softmax'))

model.summary()

# Compute the model parameters

              loss='sparse_categorical_crossentropy',
              metrics=['accuracy'])
